#include "HashTable.h"
#include <iostream>

int main(int argc, char **argv)
{
	HashTable table;

	table.SetValue("Aaron", "012346789");
	table.SetValue("Mara", "12345636205");
	table.SetValue("Fred", "012346789");
	table.SetValue("random", "012346789");
	table.SetValue("random1", "012346789");
	table.SetValue("random2", "012346789");
	table.SetValue("random3", "012346789");
	
	std::cout << table.GetValue("Mara") << std::endl; // 12345636205

	std::cout << table.HasValue("Bob") << std::endl; // false;
	std::cout << table.HasValue("Aaron") << std::endl; // true;

	std::cout << table.print() << std::endl;
}
